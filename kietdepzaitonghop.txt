local MusicGroup = TabMusic:AddLeftGroupbox("Music Player")
local currentSound = nil

local function stopMusic()
    if currentSound then
        currentSound:Stop()
        currentSound:Destroy()
        currentSound = nil
    end
end

local function playMusic(id, loop)
    stopMusic()
    currentSound = Instance.new("Sound", workspace)
    currentSound.SoundId = "rbxassetid://" .. tostring(id)
    currentSound.Looped = loop or false
    currentSound:Play()
end

MusicGroup:AddButton({
    Callback = function() stopMusic() end,
    Title = "🔇 Dừng Nhạc (Stop)"
})
